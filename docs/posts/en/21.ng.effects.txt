private readonly localStoreEffect = effect(() => {
  const userToken = this.authState(); // trigger
  // effect
  if (userToken === NULL_USER_TOKEN) {
    localStorage.removeItem('userToken');
  } else {
    localStorage.setItem('userToken', JSON.stringify(userToken));
  }
